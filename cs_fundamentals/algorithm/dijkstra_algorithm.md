import java.util.*;

public class DijkstraAlgorithm {
    
    public static Map<String, Integer> dijkstra(
            Map<String, List<int[]>> graph, 
            String start,
            Map<String, String> previous) {
        
        Map<String, Integer> distances = new HashMap<>();
        Set<String> visited = new HashSet<>();
        PriorityQueue<int[]> pq = new PriorityQueue<>(
            Comparator.comparingInt(a -> a[0])
        );
        
        // ì´ˆê¸°í™”
        for (String vertex : graph.keySet()) {
            distances.put(vertex, Integer.MAX_VALUE);
            previous.put(vertex, null);
        }
        distances.put(start, 0);
        pq.offer(new int[]{0, start.charAt(0)});
        
        while (!pq.isEmpty()) {
            int[] current = pq.poll();
            String currentVertex = String.valueOf((char) current[1]);
            
            if (visited.contains(currentVertex)) continue;
            visited.add(currentVertex);
            
            // ì¸ì ‘ ì •ì  ì™„í™”
            for (int[] neighbor : graph.getOrDefault(currentVertex, new ArrayList<>())) {
                String nextVertex = String.valueOf((char) neighbor[0]);
                int weight = neighbor[1];
                int newDist = distances.get(currentVertex) + weight;
                
                if (newDist < distances.get(nextVertex)) {
                    distances.put(nextVertex, newDist);
                    previous.put(nextVertex, currentVertex);
                    pq.offer(new int[]{newDist, neighbor[0]});
                }
            }
        }
        
        return distances;
    }
}
