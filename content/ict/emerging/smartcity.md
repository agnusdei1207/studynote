# 스마트시티 (Smart City)

## 핵심 인사이트 (3줄 요약)
> **ICT 기술로 도시 문제를 해결하고 시민 삶의 질을 향상**. 교통, 에너지, 안전, 행정 등 도시 서비스 지능화. 데이터 기반 의사결정이 핵심.

## 1. 개념
스마트시티는 **ICT(정보통신기술)를 활용하여 도시 인프라와 서비스를 지능화**하여 시민의 삶의 질을 향상하고 도시 운영 효율성을 높이는 도시 모델이다.

> 비유: "도시의 뇌" - 모든 것이 연결되고 최적화되는 도시

## 5. 코드 예시

```python
from dataclasses import dataclass, field
from typing import Dict, List, Optional
from datetime import datetime
from enum import Enum
import random

class ServiceType(Enum):
    TRAFFIC = "교통"
    ENERGY = "에너지"
    SAFETY = "안전"
    ENVIRONMENT = "환경"
    ADMIN = "행정"

@dataclass
class Sensor:
    """센서"""
    sensor_id: str
    sensor_type: str
    location: str
    value: float
    timestamp: datetime

@dataclass
class TrafficData:
    """교통 데이터"""
    intersection_id: str
    vehicle_count: int
    avg_speed: float
    congestion_level: str
    signal_status: str

@dataclass
class EnergyData:
    """에너지 데이터"""
    zone_id: str
    consumption: float
    generation: float
    peak_usage: float

class SmartTrafficSystem:
    """스마트 교통 시스템"""

    def __init__(self):
        self.intersections: Dict[str, TrafficData] = {}
        self.signal_cycles: Dict[str, int] = {}

    def update_traffic(self, intersection_id: str, vehicle_count: int, avg_speed: float):
        """교통 상황 업데이트"""
        # 혼잡도 계산
        if vehicle_count > 50 or avg_speed < 20:
            congestion = "혼잡"
        elif vehicle_count > 30 or avg_speed < 40:
            congestion = "보통"
        else:
            congestion = "원활"

        # 신호 주기 조정
        if congestion == "혼잡":
            signal_cycle = 120  # 긴 주기
        elif congestion == "보통":
            signal_cycle = 90
        else:
            signal_cycle = 60

        self.intersections[intersection_id] = TrafficData(
            intersection_id=intersection_id,
            vehicle_count=vehicle_count,
            avg_speed=avg_speed,
            congestion_level=congestion,
            signal_status="최적화됨"
        )
        self.signal_cycles[intersection_id] = signal_cycle

    def get_optimal_route(self, start: str, end: str) -> Dict:
        """최적 경로 안내 (시뮬레이션)"""
        # 혼잡도 기반 경로 선택
        congested = [k for k, v in self.intersections.items() if v.congestion_level == "혼잡"]

        return {
            'start': start,
            'end': end,
            'estimated_time': random.randint(10, 30),
            'avoid_intersections': congested,
            'recommendation': '최적 경로입니다'
        }

    def get_city_traffic_summary(self) -> Dict:
        """도시 교통 현황"""
        if not self.intersections:
            return {}

        total_vehicles = sum(t.vehicle_count for t in self.intersections.values())
        avg_speed = sum(t.avg_speed for t in self.intersections.values()) / len(self.intersections)

        congestion_counts = {}
        for t in self.intersections.values():
            congestion_counts[t.congestion_level] = congestion_counts.get(t.congestion_level, 0) + 1

        return {
            'total_intersections': len(self.intersections),
            'total_vehicles': total_vehicles,
            'avg_speed': round(avg_speed, 1),
            'congestion_status': congestion_counts
        }

class SmartEnergySystem:
    """스마트 에너지 시스템"""

    def __init__(self):
        self.zones: Dict[str, EnergyData] = {}
        self.grid_load: float = 0.0

    def update_consumption(self, zone_id: str, consumption: float, generation: float):
        """에너지 소비/생성 업데이트"""
        peak = max(consumption, self.zones.get(zone_id, EnergyData("", 0, 0, 0)).peak_usage)
        self.zones[zone_id] = EnergyData(zone_id, consumption, generation, peak)
        self.grid_load = sum(z.consumption for z in self.zones.values())

    def optimize_distribution(self) -> Dict:
        """에너지 분배 최적화"""
        total_generation = sum(z.generation for z in self.zones.values())
        total_consumption = sum(z.consumption for z in self.zones.values())

        surplus = total_generation - total_consumption

        recommendations = []
        if surplus > 0:
            recommendations.append(f"잉여 전력 {surplus:.1f}kWh를 저장합니다")
        elif surplus < 0:
            recommendations.append(f"전력 부족 {-surplus:.1f}kWh를 그리드에서 공급받습니다")

        return {
            'total_generation': total_generation,
            'total_consumption': total_consumption,
            'surplus': surplus,
            'grid_load': self.grid_load,
            'recommendations': recommendations
        }

class SmartCityDashboard:
    """스마트시티 통합 대시보드"""

    def __init__(self, city_name: str):
        self.city_name = city_name
        self.traffic = SmartTrafficSystem()
        self.energy = SmartEnergySystem()
        self.sensors: List[Sensor] = []

    def collect_sensor_data(self):
        """센서 데이터 수집"""
        # 교통 센서
        for i in range(5):
            self.traffic.update_traffic(
                f"intersection_{i+1}",
                random.randint(10, 70),
                random.randint(15, 60)
            )

        # 에너지 센서
        for i in range(3):
            self.energy.update_consumption(
                f"zone_{i+1}",
                random.uniform(100, 500),
                random.uniform(50, 200)
            )

    def get_city_status(self) -> Dict:
        """도시 현황"""
        return {
            'city': self.city_name,
            'traffic': self.traffic.get_city_traffic_summary(),
            'energy': self.energy.optimize_distribution(),
            'timestamp': datetime.now().isoformat()
        }


# 사용 예시
print("=== 스마트시티 시뮬레이션 ===\n")

city = SmartCityDashboard("스마트시티")

# 데이터 수집
print("--- 센서 데이터 수집 ---")
city.collect_sensor_data()

# 교통 현황
print("\n--- 교통 현황 ---")
traffic_summary = city.traffic.get_city_traffic_summary()
print(f"교차로 수: {traffic_summary['total_intersections']}")
print(f"총 차량: {traffic_summary['total_vehicles']}")
print(f"평균 속도: {traffic_summary['avg_speed']}km/h")
print(f"혼잡도: {traffic_summary['congestion_status']}")

# 경로 안내
route = city.traffic.get_optimal_route("A", "B")
print(f"\n경로 안내: {route['start']} → {route['end']}")
print(f"예상 시간: {route['estimated_time']}분")

# 에너지 현황
print("\n--- 에너지 현황 ---")
energy_status = city.energy.optimize_distribution()
print(f"총 발전량: {energy_status['total_generation']:.1f}kWh")
print(f"총 소비량: {energy_status['total_consumption']:.1f}kWh")
for rec in energy_status['recommendations']:
    print(f"  → {rec}")
